  transactionsReceived Transaction[] @relation("Payee")

  notifications   Notification[]
  auditLogs       AuditLog[]

  userAgent       String?     // Last login UA

  @@index([email])
  @@index([phone])
  @@index([userType])
  @@index([kycStatus])
  @@index([status])
  @@index([createdAt])

// ============= PAYMENTS & LEDGER (Double-Entry) =============
model Transaction {
  id              String    @id @default(cuid())
  
  // Context
  projectId       String?
  project         Project?  @relation(fields: [projectId], references: [id])
  milestoneId     String?
  milestone       Milestone? @relation(fields: [milestoneId], references: [id])
  payAppId        String?
  lienWaiverId    String?
  
  // Parties
  payerId         String
  payer           User      @relation("Payer", fields: [payerId], references: [id])
  payeeId         String
  payee           User      @relation("Payee", fields: [payeeId], references: [id])
  
  // Type & Status
  type            TransactionType
  status          TransactionStatus @default(INITIATED)
  
  // Amounts (always in USD cents for precision)
  grossAmount     Int       // Total before fees
  platformFee     Int       @default(0)
  processingFee   Int       @default(0)
  netAmount       Int       // What payee receives
  currency        String    @default("USD")
  
  // Payment Method & External IDs
  paymentMethod   String    // STRIPE_ACH, STRIPE_CARD, UNIT_ACH, INTERNAL_LEDGER, WIRE
  stripeChargeId  String?
  stripeTransferId String?
  unitTransactionId String?
  achTraceNumber  String?
  
  // Timing
  initiatedAt     DateTime  @default(now())
  authorizedAt    DateTime?
  capturedAt      DateTime?
  settledAt       DateTime?
  failedAt        DateTime?
  
  // Escrow/Retainage
  isEscrow        Boolean   @default(false)
  isRetainage     Boolean   @default(false)
